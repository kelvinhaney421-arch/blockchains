    <script>
        // Global state
        let walletConnected = false;
        let currentAccount = null;
        let balanceVisible = true;

        // Initialize Lucide icons
        lucide.createIcons();

        // Modal content
        const modalContent = {
            buy: { title: 'Buy Crypto', description: 'Purchase cryptocurrency using your preferred payment method. Connect your wallet and choose from multiple payment options including credit card, bank transfer, and more.' },
            swap: { title: 'Swap Tokens', description: 'Exchange tokens across multiple networks with the best rates. Our aggregator finds the optimal route for your swaps across all major DEXs.' },
            earn: { title: 'Earn Rewards', description: 'Stake your tokens and earn passive income. Choose from various staking pools and earn rewards while helping secure the network.' },
            spend: { title: 'MetaMask Card', description: 'Get your MetaMask Card and spend crypto anywhere Mastercard is accepted. Convert your crypto to fiat instantly at the point of sale.' },
            send: { title: 'Send Crypto', description: 'Send cryptocurrency to any wallet address. Support for all major tokens and networks with real-time gas optimization.' }
        };

        // Wallet detection and connection
        async function connectWallet() {
            const btn = document.getElementById('connectBtn');
            const originalContent = btn.innerHTML;
            
            // Show loading state
            btn.disabled = true;
            btn.innerHTML = '<div class="loading-spinner"></div><span>CONNECTING...</span>';
            
            try {
                // Check if MetaMask is installed
                if (typeof window.ethereum === 'undefined') {
                    showWalletNotFoundError();
                    btn.innerHTML = originalContent;
                    btn.disabled = false;
                    return;
                }

                // Update status to connecting
                updateWalletStatus('connecting', 'Connecting to MetaMask...');

                // Request account access
                const accounts = await window.ethereum.request({
                    method: 'eth_requestAccounts'
                });

                if (accounts.length > 0) {
                    walletConnected = true;
                    currentAccount = accounts[0];
                    
                    // Update UI for connected state
                    document.body.classList.add('wallet-connected');
                    btn.innerHTML = '<i data-lucide="check-circle" width="16" height="16"></i><span>CONNECTED</span>';
                    btn.style.background = '#22c55e';
                    
                    updateWalletStatus('connected', `Connected: ${currentAccount.substring(0, 6)}...${currentAccount.substring(38)}`);
                    
                    // Get network info
                    await getNetworkInfo();
                    
                    // Get balance
                    await updateBalance();
                    
                    showToast('MetaMask connected successfully!', 'success');
                    
                    // Re-initialize icons
                    lucide.createIcons();
                }
            } catch (error) {
                console.error('Connection failed:', error);
                btn.innerHTML = originalContent;
                btn.disabled = false;
                updateWalletStatus('disconnected', 'Connection failed');
                showToast('Failed to connect wallet. Please try again.', 'error');
            }
        }

        function showWalletNotFoundError() {
            const heroSection = document.querySelector('.hero-section');
            const existingError = document.querySelector('.wallet-not-found');
            
            if (!existingError) {
                const errorDiv = document.createElement('div');
                errorDiv.className = 'wallet-not-found';
                errorDiv.innerHTML = `
                    <strong>MetaMask Not Detected</strong><br>
                    Please install MetaMask to connect your wallet. 
                    <a href="https://metamask.io/download/" target="_blank">Download here</a>
                `;
                heroSection.appendChild(errorDiv);
            }
            
            showToast('MetaMask not detected. Please install MetaMask extension.', 'error');
        }

        function updateWalletStatus(status, message) {
            const statusEl = document.getElementById('walletStatus');
            statusEl.className = `wallet-status ${status}`;
            
            let icon = 'alert-circle';
            if (status === 'connected') icon = 'check-circle';
            if (status === 'connecting') icon = 'loader';
            
            statusEl.innerHTML = `
                <i data-lucide="${icon}" width="16" height="16"></i>
                <span>${message}</span>
            `;
            lucide.createIcons();
        }

        async function getNetworkInfo() {
            try {
                const chainId = await window.ethereum.request({ method: 'eth_chainId' });
                console.log('Connected to network:', chainId);
            } catch (error) {
                console.error('Failed to get network info:', error);
            }
        }

        async function updateBalance() {
            if (!currentAccount) return;
            
            try {
                const balance = await window.ethereum.request({
                    method: 'eth_getBalance',
                    params: [currentAccount, 'latest']
                });
                
                // Convert wei to ETH (simplified)
                const ethBalance = parseInt(balance, 16) / Math.pow(10, 18);
                const usdValue = ethBalance * 2400; // Approximate ETH price
                
                document.getElementById('portfolioValue').textContent = `$${usdValue.toFixed(2)}`;
                document.getElementById('balanceAmount').textContent = `$${usdValue.toFixed(2)}`;
            } catch (error) {
                console.error('Failed to get balance:', error);
            }
        }

        // Listen for account changes
        if (typeof window.ethereum !== 'undefined') {
            window.ethereum.on('accountsChanged', function (accounts) {
                if (accounts.length === 0) {
                    // User disconnected
                    walletConnected = false;
                    currentAccount = null;
                    document.body.classList.remove('wallet-connected');
                    updateWalletStatus('disconnected', 'Wallet Disconnected');
                    document.getElementById('connectBtn').innerHTML = '<i data-lucide="wallet" width="16" height="16"></i><span>CONNECT METAMASK</span>';
                    document.getElementById('connectBtn').style.background = '#f6851b';
                    showToast('Wallet disconnected', 'info');
                } else {
                    // Account changed
                    currentAccount = accounts[0];
                    updateWalletStatus('connected', `Connected: ${currentAccount.substring(0, 6)}...${currentAccount.substring(38)}`);
                    updateBalance();
                    showToast('Account changed', 'info');
                }
                lucide.createIcons();
            });

            window.ethereum.on('chainChanged', function (chainId) {
                showToast('Network changed', 'info');
                getNetworkInfo();
            });
        }

        // Sidebar functionality
        function toggleSidebar() {
            const sidebar = document.getElementById('sidebar');
            sidebar.classList.toggle('open');
        }

        function setActiveNav(element) {
            document.querySelectorAll('.nav-item').forEach(item => item.classList.remove('active'));
            element.classList.add('active');
        }

        function switchAccount(element, avatar, name, balance) {
            document.querySelectorAll('.account-item').forEach(item => {
                item.classList.remove('active');
                const checkIcon = item.querySelector('i[data-lucide="check-circle"]');
                if (checkIcon) checkIcon.remove();
            });
            
            element.classList.add('active');
            const checkIcon = document.createElement('i');
            checkIcon.setAttribute('data-lucide', 'check-circle');
            checkIcon.setAttribute('width', '16');
            checkIcon.setAttribute('height', '16');
            checkIcon.setAttribute('color', '#22c55e');
            element.appendChild(checkIcon);
            
            lucide.createIcons();
            showToast(`Switched to ${name}`, 'success');
        }

        // Modal functionality
        function showModal(action) {
            const content = modalContent[action] || modalContent.buy;
            document.getElementById('modal-title').textContent = content.title;
            document.getElementById('modal-description').textContent = content.description;
            document.getElementById('modal').classList.add('show');
            document.body.style.overflow = 'hidden';
        }

        function closeModal() {
            document.getElementById('modal').classList.remove('show');
            document.body.style.overflow = 'auto';
        }

        // Toast functionality
        function showToast(message, type = 'info') {
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.textContent = message;
            toast.onclick = () => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 300);
            };
            
            document.body.appendChild(toast);
            
            setTimeout(() => toast.classList.add('show'), 100);
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => toast.remove(), 300);
            }, 4000);
        }

        // Balance visibility toggle
        function toggleBalanceVisibility() {
            balanceVisible = !balanceVisible;
            const balanceEl = document.getElementById('balanceAmount');
            const eyeIcon = document.querySelector('.balance-section i[data-lucide="eye"], .balance-section i[data-lucide="eye-off"]');
            
            if (balanceVisible) {
                balanceEl.textContent = '$10,027.00';
                eyeIcon.setAttribute('data-lucide', 'eye');
            } else {
                balanceEl.textContent = '••••••';
                eyeIcon.setAttribute('data-lucide', 'eye-off');
            }
            lucide.createIcons();
        }

        // Event listeners
        document.addEventListener('keydown', (e) => {
            if (e.key === 'Escape') closeModal();
        });

        document.getElementById('modal').addEventListener('click', (e) => {
            if (e.target.classList.contains('modal')) closeModal();
        });

        // Close sidebar on outside click (mobile)
        document.addEventListener('click', (e) => {
            const sidebar = document.getElementById('sidebar');
            const mobileBtn = document.querySelector('.mobile-menu-btn');
            
            if (window.innerWidth <= 1024 && 
                !sidebar.contains(e.target) && 
                !mobileBtn.contains(e.target) &&
                sidebar.classList.contains('open')) {
                sidebar.classList.remove('open');
            }
        });

        // Smooth scrolling for navigation
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

        // Add interactive animations for platform cards
        document.querySelectorAll('.platform-card').forEach(card => {
            card.addEventListener('mouseenter', function() {
                this.style.transform = 'translateY(-8px) scale(1.02)';
            });
            
            card.addEventListener('mouseleave', function() {
                this.style.transform = 'translateY(0) scale(1)';
            });
        });

        // Initialize on load
        document.addEventListener('DOMContentLoaded', function() {
            lucide.createIcons();
            
            // Auto-connect if previously connected
            if (typeof window.ethereum !== 'undefined' && localStorage.getItem('walletConnected') === 'true') {
                setTimeout(() => {
                    connectWallet();
                }, 1000);
            }
            
            // Show welcome message
            setTimeout(() => {
                showToast('Welcome to MetaMask Platform! Connect your wallet to get started.', 'info');
            }, 2000);
        });

        // Re-initialize icons periodically to catch any dynamic content
        setInterval(() => lucide.createIcons(), 2000);

        // Save connection state
        window.addEventListener('beforeunload', () => {
            localStorage.setItem('walletConnected', walletConnected.toString());
        });

        // Performance optimization: Lazy load images and animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '50px'
        };

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                }
            });
        }, observerOptions);

        // Observe elements for animation
        document.querySelectorAll('.stat-card, .action-card, .platform-card').forEach(el => {
            el.style.opacity = '0';
            el.style.transform = 'translateY(20px)';
            el.style.transition = 'opacity 0.6s ease, transform 0.6s ease';
            observer.observe(el);
        });
    </script>
